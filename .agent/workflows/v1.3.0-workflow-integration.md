## Technical Specifications

### Database Schema

```typescript
// Project
interface Project {
  id: string;
  name: string;
  description: string;
  createdAt: number;
  modifiedAt: number;
  tags: string[];
  status: 'active' | 'archived';
  settings: ProjectSettings;
}

// History Entry
interface HistoryEntry {
  id: string;
  projectId: string;
  timestamp: number;
  prompt: string;
  metadata: PromptMetadata;
  tags: string[];
  favorite: boolean;
}

// Database Indexes
- projects: id, createdAt, modifiedAt, status
- history: id, projectId, timestamp, tags
- templates: id, projectId, category
```

### API Export Format

```json
{
  "version": "1.0",
  "timestamp": "2026-02-09T21:33:14Z",
  "project": {
    "id": "project-123",
    "name": "My Film Project"
  },
  "prompts": [
    {
      "id": "prompt-456",
      "type": "video",
      "content": "...",
      "metadata": {
        "style": "cinematic",
        "camera": "wide-angle",
        "duration": 5
      }
    }
  ],
  "_links": {
    "self": "/api/v1/projects/project-123/prompts",
    "project": "/api/v1/projects/project-123"
  }
}
```
